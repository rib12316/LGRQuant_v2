```python
"""验证 marlin 安装 + 探测 API 签名。"""
import torch
import marlin

def test_marlin_basic():
    print("marlin version:", getattr(marlin, "__version__", "unknown"))
    
    k, n = 4096, 4096
    gs = 128
    
    # 构造随机 fp16 权重 + scale
    w = torch.randn(n, k, dtype=torch.float16, device="cuda")
    s = torch.ones(k // gs, n, dtype=torch.float16, device="cuda")
    
    # 创建 marlin layer
    layer = marlin.Layer(k, n, groupsize=gs).cuda()
    
    # 探测 pack API
    temp_linear = torch.nn.Linear(k, n, bias=False)
    temp_linear.weight.data = w
    try:
        layer.pack(temp_linear, s)
        print("pack() OK")
    except TypeError as e:
        print("pack() failed:", e)
        # fallback: 尝试不同参数顺序
        try:
            layer.pack(linear=temp_linear, scales=s)
            print("pack(linear=..., scales=...) OK")
        except Exception as e2:
            print("all pack variants failed:", e2)
            raise
    
    # 探测 mul API
    x = torch.randn(2, k, dtype=torch.float16, device="cuda")
    out = torch.empty(2, n, dtype=torch.float16, device="cuda")
    try:
        marlin.mul(x, layer.B, out, layer.s, layer.workspace)
        print("mul() OK, out shape:", out.shape)
    except Exception as e:
        print("mul() failed:", e)
        raise
    
    # 数值校验：mul 输出应与 fp16 matmul 接近
    ref = torch.matmul(x, w.t())
    diff = (out - ref).abs().max().item()
    print(f"max diff vs fp16 matmul: {diff}")
    assert diff < 0.5, f"marlin GEMM accuracy too poor: {diff}"
    print("SMOKE PASSED")

if __name__ == "__main__":
    test_marlin_basic()
```

```python
# decoupleQ/linear_w4a16.py
"""Marlin-backed W4A16 Linear (sym only, group_size in {128, -1}).

与 linear_w2a16.py 1:1 对应：
  - make_qw4_linear() 在 inference_test.py 的加载流程中装配 weight/scale/bias
  - _prewarm_w4a16() 在加载阶段一次性做 int8 -> marlin packed 转换
  - forward() 直接调 marlin.mul，输入/输出统一 fp16
"""
import torch
import torch.nn as nn
import marlin


class LinearW4A16(nn.Module):
    def __init__(self, in_features: int, out_features: int, bias: bool, group_size: int):
        super().__init__()
        # group_size=-1 表示 per-channel，Marlin 用 groupsize=k
        self.k = in_features
        self.n = out_features
        self.with_bias = bias is not None
        self.group_size = group_size

        # 以下字段在 make_qw4_linear / _prewarm_w4a16 中被 setattr
        self.weight = None          # int8 [in, out] (来自 ckpt)
        self.qweight = None         # marlin packed int4
        self.scales = None          # fp16 [groups, out]
        self.workspace = None       # marlin GEMM staging buffer
        self.bias = None            # fp16 [out]
        self.weight_processed = False

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        assert self.weight_processed, "LinearW4A16: weights not packed yet"
        orig_dtype = x.dtype
        x_fp16 = x if x.dtype == torch.float16 else x.to(torch.float16)
        out = torch.empty(x_fp16.shape[:-1] + (self.n,),
                          dtype=torch.float16, device=x_fp16.device)
        marlin.mul(x_fp16, self.qweight, out, self.scales, self.workspace)
        if self.bias is not None:
            out = out + self.bias
        return out if orig_dtype == torch.float16 else out.to(orig_dtype)


def convert_int8_to_marlin(int8_w: torch.Tensor, fp32_scale: torch.Tensor,
                           group_size: int):
    """把 sym int8 ckpt 权重转换为 marlin packed 格式。

    Args:
        int8_w:      [in_features, out_features] int8 (来自 ckpt weight.t())
        fp32_scale:  [groups, out_features] fp32 (来自 ckpt qscale.t())
        group_size:  128 或 -1

    Returns:
        (qweight, scales, workspace) 可直接赋给 LinearW4A16
    """
    k, n = int8_w.shape
    gs = group_size if group_size > 0 else k  # per-channel: gs=k
    assert k % gs == 0, f"k={k} 不能被 group_size={gs} 整除"

    # 1) 反量化为 fp16 dense weight（marlin.pack 需要输入 fp16 Linear）
    scale_expanded = fp32_scale.to(torch.float16).repeat_interleave(gs, dim=0)  # [k, n]
    w_fp16 = int8_w.to(torch.float16) * scale_expanded  # [k, n]

    # 2) 构造临时 nn.Linear（weight 布局 [out, in]）
    temp_linear = nn.Linear(k, n, bias=False)
    temp_linear.weight.data = w_fp16.t().contiguous()

    # 3) marlin pack
    scales_fp16 = fp32_scale.to(torch.float16).contiguous()  # [groups, n] fp16
    mlayer = marlin.Layer(k, n, groupsize=gs).cuda()
    mlayer.pack(temp_linear, scales_fp16)

    return mlayer.B, mlayer.s, mlayer.workspace


def _prewarm_w4a16(model):
    """对所有 LinearW4A16 做一次性 marlin packing。"""
    n = 0
    for m in model.modules():
        if isinstance(m, LinearW4A16) and not m.weight_processed:
            assert m.weight is not None and m.scales is not None
            m.qweight, m.scales, m.workspace = convert_int8_to_marlin(
                m.weight, m.scales, m.group_size
            )
            if m.bias is not None:
                m.bias = m.bias.to(torch.float16).contiguous()
            m.weight_processed = True
            n += 1
    print(f"  [opt] pre-warmed {n} LinearW4A16 layers")
```

```python
"""LinearW4A16 单元测试：对比 marlin GEMM vs 手写 dequant + matmul。"""
import sys, os
sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

import torch
from decoupleQ.linear_w4a16 import LinearW4A16, convert_int8_to_marlin


def test_linear_w4a16_cosine():
    k, n = 4096, 11008
    gs = 128
    batch = 4

    # 构造随机 sym int8 权重 + fp32 scale
    int8_w = torch.randint(-8, 8, (k, n), dtype=torch.int8, device="cuda")
    scale = torch.rand(k // gs, n, dtype=torch.float32, device="cuda") * 0.01 + 0.001

    # 1) 用 convert_int8_to_marlin 得到 packed 权重
    qw, s, ws = convert_int8_to_marlin(int8_w, scale, gs)

    # 2) 装配 LinearW4A16
    layer = LinearW4A16(k, n, bias=False, group_size=gs)
    layer.weight = int8_w
    layer.scales = scale.to(torch.float16)
    layer.qweight = qw
    layer.scales = s
    layer.workspace = ws
    layer.weight_processed = True

    # 3) 输入 fp16
    x = torch.randn(batch, k, dtype=torch.float16, device="cuda")

    # 4) Marlin 输出
    out_marlin = layer(x)

    # 5) 参考输出：dequant + matmul
    scale_expanded = scale.to(torch.float16).repeat_interleave(gs, dim=0)
    w_fp16 = int8_w.to(torch.float16) * scale_expanded
    out_ref = torch.matmul(x, w_fp16)

    # 6) 校验
    cosine = torch.nn.functional.cosine_similarity(
        out_marlin.flatten(), out_ref.flatten(), dim=0
    ).item()
    max_diff = (out_marlin - out_ref).abs().max().item()
    print(f"cosine={cosine:.6f}  max_diff={max_diff:.4f}")
    assert cosine > 0.999, f"cosine too low: {cosine}"
    assert max_diff < 0.5, f"max_diff too high: {max_diff}"
    print("PASSED")


if __name__ == "__main__":
    test_linear_w4a16_cosine()
```

```python
def make_qw4_linear(old_linear: nn.Linear, sd: dict, group_size: int, name: str):
    """把一个 nn.Linear 替换成 LinearW4A16（sym，无 zp）。"""
    in_features = old_linear.in_features
    out_features = old_linear.out_features
    has_bias = old_linear.bias is not None

    layer = LinearW4A16(in_features, out_features, has_bias, group_size)

    # weight (int8) —— ckpt 里存 [out, in]，转置为 [in, out]
    w = sd[f"{name}.weight"]
    assert w.dtype == torch.int8, f"{name}.weight expected int8, got {w.dtype}"
    layer.weight = w.t().contiguous()  # [in, out] int8

    # scale（sym 路径下只有 qscale，无 qzero）
    if f"{name}.weight_qscale" in sd:
        # ckpt: [out, groups] fp32 -> LinearW4A16 期望 [groups, out] fp16（pack 时会再转）
        scale = sd[f"{name}.weight_qscale"].cuda().to(torch.float16).t().contiguous()
        layer.scales = scale

    # bias
    if has_bias:
        if f"{name}.bias" in sd:
            b = sd[f"{name}.bias"]
        else:
            b = old_linear.bias.detach()
        layer.bias = b.cuda().to(torch.float16).contiguous()

    return layer
```

```python
def replace_linear_quantized(model, sd: dict, group_size: int, make_fn):
    """通用量化层替换：对 model.model.layers 里的 q/k/v/o/gate/up/down 做替换。"""
    layers = model.model.layers
    targets = [
        ("self_attn", "q_proj"),
        ("self_attn", "k_proj"),
        ("self_attn", "v_proj"),
        ("self_attn", "o_proj"),
        ("mlp", "gate_proj"),
        ("mlp", "up_proj"),
        ("mlp", "down_proj"),
    ]
    for i, layer in enumerate(layers):
        for parent, child in targets:
            full_name = f"model.layers.{i}.{parent}.{child}"
            if f"{full_name}.weight" not in sd:
                continue
            old = getattr(getattr(layer, parent), child)
            new = make_fn(old, sd, group_size, full_name)
            setattr(getattr(layer, parent), child, new)
    return model


def replace_linear_with_w2(model, sd: dict, group_size: int):
    """W2A16 替换（兼容旧入口）。"""
    return replace_linear_quantized(model, sd, group_size, make_qw2_linear)


def replace_linear_with_w4(model, sd: dict, group_size: int):
    """W4A16 Marlin 替换。"""
    return replace_linear_quantized(model, sd, group_size, make_qw4_linear)
```

```python
def load_true_quant_model(model_path, ckpt_path, group_size, kernel="w2",
                          fp16_main=False):
    print(f"[Load] true_quant ckpt: {ckpt_path}  kernel={kernel}")
    model = _get_model(model_path)

    sd = torch.load(ckpt_path, map_location="cpu")

    # 1) 先把非量化参数（embed/norm/lm_head/bias）灌入主干
    fp_sd = {k: v for k, v in sd.items()
             if not k.endswith(".weight_qscale")
             and not k.endswith(".weight_qzero")
             and v.dtype != torch.int8}
    missing, unexpected = model.load_state_dict(fp_sd, strict=False)
    print(f"  load_state_dict (FP part): missing={len(missing)} unexpected={len(unexpected)}")

    # 2) 用对应内核替换 q/k/v/o/gate/up/down
    if kernel == "w2":
        replace_linear_with_w2(model, sd, group_size)
    elif kernel == "w4":
        replace_linear_with_w4(model, sd, group_size)
    else:
        raise ValueError(f"unknown kernel: {kernel}")

    # 3) 主干转 fp16
    if fp16_main:
        print("  [opt] convert backbone to fp16 (remove per-layer dtype cast)")
        model = model.to(torch.float16)

    model = model.cuda()

    # 4) 预热
    if kernel == "w2":
        _prewarm_w2a16(model)
    elif kernel == "w4":
        _prewarm_w4a16(model)

    return model
```

- [ ] **Step 1: 写脚本**

```bash
#!/bin/bash
# =====================================================================
# LGRQuant 快速量化流水线（仅 Stage 1 + Stage 2）
#   支持 WBITS=2|4 切换
#   Stage 1: dq_xx  |  Stage 2: ptq1.61
# =====================================================================
set -e

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
cd "$SCRIPT_DIR"

# -------------------- 可配置参数 --------------------
BASE_MODEL="${BASE_MODEL:-/data2/llms/Qwen2.5-3b}"
GPU_ID="${GPU_ID:-0}"
WBITS="${WBITS:-4}"
S2_TRAIN_STEPS="${S2_TRAIN_STEPS:-2000}"
OUT_ROOT="${OUT_ROOT:-/data2/xx_llms/LGRQuant_fast}"

# 根据 WBITS 设置量化参数
if [ "$WBITS" = "2" ]; then
    GS=64
    ASYM_FLAG="--asym"
    QF_ASYM_FLAG="--asym"
elif [ "$WBITS" = "4" ]; then
    GS=-1                    # per-channel（group quant 不支持 sym）
    ASYM_FLAG=""
    QF_ASYM_FLAG=""
else
    echo "ERROR: WBITS must be 2 or 4"
    exit 1
fi

OUT_S1="${OUT_ROOT}/w${WBITS}/stage1"
OUT_S2="${OUT_ROOT}/w${WBITS}/stage2"
mkdir -p "$OUT_S1" "$OUT_S2"

LOG_DIR="${SCRIPT_DIR}/logs_fast"
mkdir -p "$LOG_DIR"

REPORT="${SCRIPT_DIR}/fast_summary.txt"
: > "$REPORT"

# 时间/显存结果文件
TIME_S1="${LOG_DIR}/time_s1.txt"
MEM_S1="${LOG_DIR}/mem_s1.txt"
TIME_S2="${LOG_DIR}/time_s2.txt"
MEM_S2="${LOG_DIR}/mem_s2.txt"

# conda 初始化
source "$(conda info --base)/etc/profile.d/conda.sh"

# -------------------- 工具函数 --------------------
monitor_simple() {
    local root_pid=$1
    local out_file=$2
    echo 0 > "$out_file"
    while kill -0 "$root_pid" 2>/dev/null; do
        local mem
        mem=$(nvidia-smi --query-compute-apps=pid,used_memory \
              --format=csv,noheader,nounits 2>/dev/null \
              | awk -F',' -v p="$root_pid" \
                   '{gsub(/ /,""); if ($1 == p) print $2}')
        local cur
        cur=$(cat "$out_file" 2>/dev/null || echo 0)
        if [ "${mem:-0}" -gt "${cur:-0}" ]; then
            echo "$mem" > "$out_file"
        fi
        sleep 0.5
    done
}

# =====================================================================
# Stage 1 — ARQ + 逐块训练
# =====================================================================
echo ">>> [Stage 1] ARQ + 逐块训练  (wbits=${WBITS}, gs=${GS})"
rm -f "$TIME_S1" "$MEM_S1"

conda activate dq_xx
export CUDA_VISIBLE_DEVICES=${GPU_ID}

S1_LOG="${LOG_DIR}/stage1_w${WBITS}.log"
: > "$S1_LOG"

python3 -u "${SCRIPT_DIR}/llama.py" \
    --model "$BASE_MODEL" \
    --dataset c4 \
    --true-sequential \
    --act-order \
    --new-eval \
    --wbits "$WBITS" \
    --group-size "$GS" \
    --nsamples 128 \
    --max-iter-num 4 \
    --iters-before-round 200 \
    --inner-iters-for-round 5 \
    --round-fn gptq \
    --blockwise-minimize-epoch 4 \
    --blockwise-minimize-lr 1.0e-5 \
    --finetune-scale-mode layerwise \
    --save \
    --save_true_only \
    --out_path "$OUT_S1" \
    ${ASYM_FLAG} >> "$S1_LOG" 2>&1 &
PID1=$!

monitor_simple "$PID1" "$MEM_S1"
wait "$PID1"
S1_EXIT=$?

if [ "$S1_EXIT" -ne 0 ]; then
    echo "ERROR: Stage 1 失败，退出码=$S1_EXIT"
    echo "日志: $S1_LOG"
    conda deactivate
    exit 1
fi

# 提取时间
grep -E '^time cost [0-9.]+, model\.' "$S1_LOG" \
    | awk '{s+=$3} END {printf "%.0f", s+0}' > "$TIME_S1"

conda deactivate
echo "    Stage 1 完成 → $OUT_S1/true_quant.pth"
echo "        时间: $(cat "$TIME_S1")s  显存: $(cat "$MEM_S1")MB"

# =====================================================================
# Stage 2 — E2E 蒸馏
# =====================================================================
echo ">>> [Stage 2] E2E 蒸馏  (train_steps=${S2_TRAIN_STEPS})"
rm -f "$TIME_S2" "$MEM_S2"

if [ ! -f "$OUT_S1/true_quant.pth" ]; then
    echo "ERROR: 未找到 $OUT_S1/true_quant.pth, 跳过 Stage 2"
    exit 1
fi

conda activate ptq1.61
export CUDA_VISIBLE_DEVICES=${GPU_ID}

S2_LOG="${LOG_DIR}/stage2_w${WBITS}.log"
: > "$S2_LOG"

python3 -u "${SCRIPT_DIR}/quant_finetune.py" \
    --model_id "$BASE_MODEL" \
    --true_quant_path "$OUT_S1/true_quant.pth" \
    --out_path "$OUT_S2" \
    --train_steps "$S2_TRAIN_STEPS" \
    --group_size "$GS" \
    --use_distill \
    --distill_loss kl_top \
    --dataset c4 \
    --nsamples 1000 \
    ${QF_ASYM_FLAG} >> "$S2_LOG" 2>&1 &
PID2=$!

monitor_simple "$PID2" "$MEM_S2"
wait "$PID2"
S2_EXIT=$?

if [ "$S2_EXIT" -ne 0 ]; then
    echo "ERROR: Stage 2 失败，退出码=$S2_EXIT"
    echo "日志: $S2_LOG"
    conda deactivate
    exit 1
fi

# 提取时间
grep -oE "'train_runtime': *[0-9.]+" "$S2_LOG" \
    | tail -n1 | grep -oE '[0-9.]+' | awk '{printf "%.0f", $1+0}' > "$TIME_S2"

conda deactivate
echo "    Stage 2 完成 → $OUT_S2/true_quant.pth"
echo "        时间: $(cat "$TIME_S2")s  显存: $(cat "$MEM_S2")MB"

# =====================================================================
# 汇总
# =====================================================================
{
echo "====================================="
echo "    LGRQuant Fast 流水线汇总"
echo "    WBITS=${WBITS}  $(date '+%Y-%m-%d %H:%M:%S')"
echo "====================================="
echo ""
echo "模型: ${BASE_MODEL}"
echo "Stage 1 ckpt: $OUT_S1/true_quant.pth"
echo "Stage 2 ckpt: $OUT_S2/true_quant.pth"
echo ""
echo "详细日志: ${LOG_DIR}"
} >> "$REPORT"

cat "$REPORT"
```

```bash
#!/usr/bin/env bash
# W4 推理验收脚本：解析 inference_test.py 输出，判定 M2 PASS/FAIL
set -e

LOG_FILE="${1:-/tmp/w4_run.log}"

if [ ! -f "$LOG_FILE" ]; then
    echo "ERROR: log file not found: $LOG_FILE"
    exit 1
fi

mem_ratio=$(grep -oP 'Memory ratio\s*=\s*\K[0-9.]+' "$LOG_FILE" | tail -n1)
tok_speedup=$(grep -oP 'Speedup \(token\)\s*=\s*\K[0-9.]+' "$LOG_FILE" | tail -n1)

if [ -z "$mem_ratio" ] || [ -z "$tok_speedup" ]; then
    echo "ERROR: 无法从日志中提取指标"
    echo "  请确认 inference_test.py 已完整运行"
    exit 1
fi

pass=1
if awk "BEGIN {exit !($mem_ratio > 0.50)}"; then
    echo "FAIL memory: ratio=${mem_ratio} > 0.50 (目标 ≤0.50)"
    pass=0
else
    echo "PASS memory: ratio=${mem_ratio} ≤ 0.50"
fi

if awk "BEGIN {exit !($tok_speedup < 1.10)}"; then
    echo "FAIL speed:  token_speedup=${tok_speedup}x < 1.10x (目标 ≥1.10x)"
    pass=0
else
    echo "PASS speed:  token_speedup=${tok_speedup}x ≥ 1.10x"
fi

if [ "$pass" -eq 1 ]; then
    echo "=== OVERALL PASS ==="
    exit 0
else
    echo "=== OVERALL FAIL ==="
    exit 1
fi
```
